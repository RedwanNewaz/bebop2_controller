17  // Set the number of particles. Initialize all particles to first position (based on estimates of
18  // x, y, theta and their uncertainties from GPS) and all weights to 1.

21 
22  weights.resize(num_particles);
23  particles.resize(num_particles);

28  std_z = std[2];
29  std_theta = std[3];

31  // Normal distribution for x, y and theta
32  normal_distribution<double> dist_x(x0[0], std_x); // mean is centered around the new measurement
33  normal_distribution<double> dist_y(x0[1], std_y);
34  normal_distribution<double> dist_z(x0[2], std_z);

41  Particle p;

65  // Add measurements to each particle and add random Gaussian noise.
66  // NOTE: When adding noise you may find std::normal_distribution and std::default_random_engine useful.

80

82  double new_x = p->x + delta_t * cmd.linear.x;
83  double new_y = p->y - delta_t * cmd.linear.y;
84  double new_z = p->z + delta_t * cmd.linear.z;
85  double new_theta = p->theta + delta_t * cmd.angular.z;
86 
87  // add Gaussian Noise to each measurement
88  // Normal distribution for x, y and theta
89  normal_distribution<double> dist_x(new_x, std_x);
90  normal_distribution<double> dist_y(new_y, std_y);
91  normal_distribution<double> dist_z(new_z, std_z);
92  normal_distribution<double> dist_theta(new_theta, std_theta);
93 
94  // update the particle attributes
95  p->x = dist_x(gen);
96  p->y = dist_y(gen);
97  p->z = dist_z(gen);
98  p->theta = dist_theta(gen);

124 void ParticleFilter::updateWeights(double sensor_range, const std::vector<double>& std_landmark,
125  std::vector<LandmarkObs> observations, TagMap map_landmarks) {
126  // Update the weights of each particle using a multi-variate Gaussian distribution. You can read
127  // more about this distribution here: https://en.wikipedia.org/wiki/Multivariate_normal_distribution
128  // NOTE: The observations are given in the VEHICLE'S coordinate system. Your particles are located
129  // according to the MAP'S coordinate system. You will need to transform between the two systems.
130  // Keep in mind that this transformation requires both rotation AND translation (but no scaling).
131  // The following is a good resource for the theory:
132  // https://www.willamette.edu/~gorr/classes/GeneralGraphics/Transforms/transforms2d.htm
133  // and the following is a good resource for the actual equation to implement (look at equation
134  // 3.33. Note that you'll need to switch the minus sign in that equation to a plus to account
135  // for the fact that the map's y-axis actually points downwards.)
136  // http://planning.cs.uiuc.edu/node99.html
137 
138 
139  double weights_sum = 0;
140 
141  for(int i=0; i<num_particles; ++i){
142  Particle *p = &particles[i];
143  double wt = 1.0;
144 
145  // step 1: collect valid landmarks
146  std::vector<LandmarkObs> predictions;
147  for(const auto& lm: map_landmarks.landmark_list){
148  double distance = dist(p->x, p->y, p->z, lm.x_d, lm.y_d, lm.z_d);
149  if( distance < sensor_range){ // if the landmark is within the sensor range, save it to predictions
150  predictions.push_back(LandmarkObs{lm.id_i, lm.x_d, lm.y_d, lm.z_d});
151  }
152  }
153 
154  // convert observation from vehicle's to map's coordinate system
155  std::vector<LandmarkObs> observations_map;
156 
157  for(const auto &current_obs : observations){
158  LandmarkObs transformed_obs;
159  // A yaw is a counterclockwise rotation of $ \theta about the $ z$-axis. The rotation matrix is given by
160  // http://msl.cs.uiuc.edu/planning/node102.html
161  // (cos(p->theta), -sin(p->theta), 0)
162  // (sin(p->theta), cos(p->theta), 0)
163  // (0, 0, 1)
164 
165  transformed_obs.x = (current_obs.x * cos(p->theta)) - (current_obs.y * sin(p->theta)) + p->x;
166  transformed_obs.y = (current_obs.x * sin(p->theta)) + (current_obs.y * cos(p->theta)) + p->y;
167  transformed_obs.z = current_obs.z + p->z;
168  transformed_obs.id = current_obs.id;
169  observations_map.push_back(transformed_obs);
170 
171  }
172 
173  // step 3: find landmark index for each observation
174  // find the predicted measurement that is closest to each observed measurement and assign
175  // the observed measurement to this particular landmark
176  dataAssociation(predictions, observations_map);
177  // step 4: compute the particle's weight:
178  // update weights using Multivariate Gaussian Distribution
179  // implement multivariate gaussian
180  for(const auto& obs_m: observations_map){
181  wt *= mvn_[obs_m.id]->pdf(obs_m);
182  weights_sum += wt;
183  p->weight = wt;
184  }
185  }
186  // normalize weights to bring them in (0, 1]
187  for (int i = 0; i < num_particles; i++) {
188  Particle *p = &particles[i];
189  p->weight /= weights_sum;
190  weights[i] = p->weight;
191  }
192 }

196  // NOTE: You may find std::discrete_distribution helpful here.

198 
199  default_random_engine gen;
