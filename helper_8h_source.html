40class StateSpace{
41public:
42 StateSpace()
43 {
44 states_.resize(STATE_DIM);
45 std::fill(states_.begin(), states_.end(), 0.0);
46 states_[2] = 1.0;
47 }
48
49 void operator()(std::vector<double>& state)
50 {
51// state.clear();
52 std::copy(states_.begin(), states_.end(), state.begin());
53
54 std::normal_distribution disturbance(0.0, sqrt(NOISE));
55
56 for (int i = 0; i < state.size(); ++i) {
57
58 state[i] += disturbance(rd_);
59 }
60
61 }
62
63 void add_control(const Twist* msg)
64 {
65 states_[0] += DT * msg->linear.x;
66 states_[1] -= DT * msg->linear.y;
67 states_[2] += DT * msg->linear.z;
68 states_[3] += DT * msg->angular.z;
69 }
70
71private:
72 std::vector<double> states_;
73 const size_t STATE_DIM = 4;
74 const double DT = 0.03;
75 const double NOISE = 0.015;
76 std::random_device rd_;
77};
