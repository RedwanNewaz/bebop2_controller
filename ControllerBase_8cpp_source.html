168  double ControllerBase::goal_distance(const std::vector<double> &X, const std::vector<double> &setPoints) {
169  double error = 0;
170  for (int i = 0; i < X.size(); ++i) {
171  double e = X[i] - setPoints[i];
172  error += e * e;
