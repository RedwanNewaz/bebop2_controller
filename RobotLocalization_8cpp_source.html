40  void RobotLocalization::obs_to_nav_msg(const std::vector<double> &obs, nav_msgs::Odometry &msg) {
41  // this function is responsible to update the ekf from robot_localization package
42  msg.header.frame_id = "mapDummy";
43  msg.header.stamp = ros::Time::now();
44  msg.pose.pose.position.x = obs[0];
45  msg.pose.pose.position.y = obs[1];
46  msg.pose.pose.position.z = obs[2];
47  // compute yaw
48  tf::Quaternion q;
49  q.setRPY(0, 0, obs[3]);
50  msg.pose.pose.orientation.x = q.x();
51  msg.pose.pose.orientation.y = q.y();
52  msg.pose.pose.orientation.z = q.z();
53  msg.pose.pose.orientation.w = q.w();
54  // update velocities
55  msg.twist.twist.linear.x = state_[4];
56  msg.twist.twist.linear.y = state_[5];
57  msg.twist.twist.linear.z = state_[6];
